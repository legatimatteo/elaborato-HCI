  \begin{tabular}{|>{\raggedright\bfseries\arraybackslash}p{3cm}|>{\raggedright\arraybackslash}p{11.5cm}|}

    \rowcolor{gray!30}
    \multicolumn{2}{|c|}{\textbf{CHECKLIST}} \\
    \hline
    Prima di accogliere l'utente &
    \vspace{-8pt}
    \begin{itemize}
      \item attivare Click counter online (link per Chrome: https://chromewebstore.google.com/detail/click-counter-online/kldnchgjdegapeicpdpnocopljohlgih) per il conteggio del numero di click
      \item Per versione A: eseguire in console
      \item Per versione B: eseguire in console
    \end{itemize}
    \vspace{-6pt} \\
    \hline
    Prima dell'esperimento &
    \vspace{-8pt}
    \begin{itemize}
      \item salutare l'utente e ringraziarlo per la disponibilità
      \item sottolineare che l'esperimento è volto alla valutazione del sito
      \item garantire la riservatezza dei dati richiesti e dei risultati
      \item descrivere la struttura generale dell'esperimento
      \item sottolineare che non si potranno chiedere consigli durante l'esperimento
      \item indicare la possibilità di abbandonare il task se reputato troppo difficile
      \item far compilare il questionario personale, ribadendo la riservatezza dei dati
      \item consegnare all'utente il modulo dei compiti
      \item illustrare brevemente il contenuto delle pagine della parte dedicata all'amministratore, descrivendo solo l'utilità generale di ciascuna pagina senza dare informazioni sui singoli elementi che la compongono
    \end{itemize}
    \vspace{-6pt} \\
    \hline
    Durante l'esperimento &
    \vspace{-8pt}
    \begin{itemize}
      \item tenere conto del numero di errori e del tempo in cui viene eseguito un singolo compito
      \item se l'utente chiede aiuto, spiegare cortesemente che non è possibile fornire consigli
    \end{itemize}
    \vspace{-6pt} \\
    \hline
    Dopo l'esperimento &
    \vspace{-8pt}
    \begin{itemize}
      \item chiedere opinioni riguardo al sito in generale (cosa è piaciuto di più, cosa di meno, cosa cambierebbe, cosa aggiungerebbe)
      \item ringraziare l'utente per la disponibilità, sottolineando l'importanza del suo contributo
      \item remunerarlo (offrendogli un caffè o un cioccolatino)
    \end{itemize}
    \vspace{-6pt} \\
    \hline
  \end{tabular}
\end{table}
